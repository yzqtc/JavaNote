  • application.properties
  • #database 数据源
    spring.datasource.url=jdbc:mysql://192.168.31.152:3306/springboot?useUnicode=true&characterEncoding=utf8&serverTimezone=UTC
    spring.datasource.username=root
    spring.datasource.password=root
    spring.datasource.driver-class-name=com.mysql.cj.jdbc.Driver
     
    #tomcat端口
    server.port=8080
     
    #整合mybatis
    #起别名,省略写mybatis的xml中的resultType的全路径
    mybatis.mapper-locations: classpath:mapper/*.xml
    #扫描(配置xml模式使用)
    mybatis.type-aliases-package=com.enjoyall.model
     
    #thymeleaf的编码
    Spring.thymeleaf.encoding=utf-8
    #热部署静态文件
    spring.thymeleaf.cache=false
    #使用html5的标准
    spring.thymeleaf.mode=HTML5
    #html的路径
    spring.thymeleaf.prefix=classpath:/templates/

  • dao
  • @Mapper
    public interface UserDao {
    List selectUser();
    }
  • server
  • public interface UserServer {
    List<User> selectUser();
    }
  • serverimpl
  • @Service
    public class UserServerImpl implements UserServer {
    @Autowired
    private UserDao udao;
    public List<User> selectUser() {
    return udao.selectUser();
    }
    }
  • controller
  • @RestController
    @RequestMapping("/user")
    public class UserController {
    @Autowired
    private UserServerImpl usi;
    @RequestMapping("")
    public String index(){
    return "index.html";
    }
    @RequestMapping("select")
    public List<User> select(){
    return usi.selectUser();
    }
    }
